  if (np === null) return null;

  <name>Task 2: src/loop.ts — Recursive-setTimeout-Loop mit AbortController + Try/Catch</name>
  <files>src/loop.ts</files>
  <action>
Lege `src/loop.ts` neu an. Exportiere `runLoop(sink, getNowPlaying, abort)`.

**Constraints (LOOP-01, LOOP-02, LOOP-03):**
- Recursive `setTimeout`, NICHT `setInterval` (Pitfall 8 + 14: setInterval überlappt bei langem Tick, setTimeout-Recursion garantiert sequentielle Ticks und überlebt Sleep/Wake besser).
- AbortController-`signal.aborted` prüft am Tick-Anfang UND vor `setTimeout(tick, 10_000)` — kein neuer Tick wird mehr registriert nach abort.
- Try/Catch um JEDEN Tick. Single Source- oder Sink-Fehler crasht den Daemon nicht (LOOP-03).
- Polling-Intervall `10_000` ms hardcoded (kein Env-Var in v1, siehe 03-CONTEXT Decisions).
- Track-Diff via `nowPlayingKey` aus `src/diff.js`. `lastKey` wird in Closure gehalten.

**Implementierung:**

```typescript
import type { GatherSink } from "./sink/gather.js";
import type { NowPlayingSource, NowPlaying } from "./sources/types.js";
import { nowPlayingKey } from "./diff.js";
import { log } from "./logger.js";

const POLL_INTERVAL_MS = 10_000;

/**
 * Polling-Loop (LOOP-01..03).
 *
 * - LOOP-01: Recursive `setTimeout` mit AbortController (NICHT `setInterval`).
 *   `setInterval` würde bei langen Ticks überlappen (Pitfall 8) und nach
 *   macOS-Sleep/Wake unsauber driften (Pitfall 14). Recursion garantiert
 *   sequentielle Ticks und sauberen Cancel via abort.signal.
 * - LOOP-02: Track-Diff via Composite-Key. Bei gleichem Key kein redundantes
 *   `setStatus` — Idempotenz schont Gather-Bandbreite und Logs.
 * - LOOP-03: Try/Catch um jeden Tick. Source-Chain wirft by contract zwar nicht
 *   (SRC-05 Layered Defense), aber Belt-and-suspenders gegen unerwartete
 *   Sink-Errors oder Modul-Lade-Fehler.
 *
 * `runLoop` ist NICHT `async` — der erste Tick wird sofort fire-and-forget
 * gestartet (`void tick()`), damit der Caller in `index.ts` direkt die Signal-
 * Handler registrieren kann, ohne auf das erste Polling-Ergebnis zu warten.
 */
export function runLoop(
  sink: GatherSink,
  getNowPlaying: NowPlayingSource,
  abort: AbortController,
): void {
  let lastKey: string | null = null;
  let timer: NodeJS.Timeout | null = null;

  const tick = async (): Promise<void> => {
    if (abort.signal.aborted) return;

    try {
      const np: NowPlaying = await getNowPlaying();
      const key = nowPlayingKey(np);

      if (key !== lastKey) {
        log.info({ from: lastKey, to: key }, "[loop] track changed");
        if (np === null) {
          sink.clearStatus();
        } else {
          sink.setStatus(np);
        }
        lastKey = key;
      }
    } catch (err) {
      // LOOP-03: Tick-Fehler isolieren, nicht crashen.
      log.error({ err }, "[loop] tick failed");
    }

    // Nach Tick erneut Abort prüfen — nichts mehr neu schedulen, wenn Shutdown läuft.
    if (!abort.signal.aborted) {
      timer = setTimeout(() => {
        void tick();
      }, POLL_INTERVAL_MS);
    }
  };

  // Optional: AbortSignal-Listener cleart pending Timer beim Shutdown,
  // damit Node nicht noch 10s auf den nächsten Tick wartet.
  abort.signal.addEventListener("abort", () => {
    if (timer !== null) {
      clearTimeout(timer);
      timer = null;
    }
  }, { once: true });

  // Erster Tick sofort, fire-and-forget.
  void tick();
}
```

**Wichtig:**
- `import type` für `GatherSink`, `NowPlayingSource`, `NowPlaying`.
- Logger-Import aus `./logger.js`.
- `void tick()` statt `tick()` (kein Promise-Floating, ESLint-no-floating-promises-konform — auch ohne ESLint guter Stil).
- `setTimeout`-Handle festhalten und in `abort`-Listener clearen — sonst hält Node nach abort noch bis zu 10s auf den pending Timer (Pitfall 14: kein Doppel-Tick durch Drift).
- `{ once: true }` auf dem abort-Listener — Memory-Leak-Prävention bei langlaufendem Daemon (Pitfall: setInterval-Closure / Performance-Trap "Memory-Leak via setInterval-Closure").
- KEINE `setInterval`, KEINE `Promise.race` für Polling-Timeout (das ist Sink-Disconnect-Sache, nicht Loop-Sache; `getNowPlaying` ist already-defensive aus Phase 2).

**Implementierung:**

```typescript
// LOOP-05/Pitfall 4: Polyfill als ALLERERSTER Import, vor jedem Modul, das
// gather-game-client transitiv lädt. gather.ts importiert setup-ws.js bereits
// selbst — defensive Doppel-Import ist günstig (idempotent, setzt globalThis nur einmal).
import "./setup-ws.js";

import pino from "pino";
import { config } from "./config.js";
import { log } from "./logger.js";
import { GatherSink } from "./sink/gather.js";
import { getNowPlaying } from "./sources/chain.js";
import { runLoop } from "./loop.js";

const SHUTDOWN_TIMEOUT_MS = 5_000;

// pino.final liefert einen sync-flushed Logger — letzte Log-Zeile landet
// auch bei sofortigem process.exit auf Disk. In Phase 4 unter launchd
// (asynchroner File-Sink) absolut kritisch, in Phase 3 (stderr/stdout
// synchron) defensiv aber harmlos.
const finalLog = pino.final(log);

async function shutdown(
  signal: string,
  abort: AbortController,
  sink: GatherSink,
): Promise<void> {
  log.info({ signal }, "[shutdown] received");
  abort.abort();

  try {
    await Promise.race([
      (async () => {
        try {
          sink.clearStatus();
        } catch (err) {
          log.warn({ err }, "[shutdown] clearStatus threw");
        }
        await sink.disconnect();
      })(),
      new Promise<never>((_, rej) =>
        setTimeout(
          () => rej(new Error("shutdown timeout")),
          SHUTDOWN_TIMEOUT_MS,
        ),
      ),
    ]);
    log.info("[shutdown] cleanup complete");
  } catch (err) {
    // 5s-Timeout oder anderer Fehler: trotzdem mit exit(0) raus.
    // launchd in Phase 4 darf das NICHT als Crash werten und Restart auslösen.
    log.warn({ err }, "[shutdown] cleanup did not complete in 5s");
  }

  process.exit(0);
}

async function main(): Promise<void> {
  log.info("[daemon] starting gatherAppleMusicBridge");

  const sink = new GatherSink(config.GATHER_SPACE_ID, config.GATHER_API_KEY);
  await sink.connect();
  log.info("[daemon] sink connected");

  const abort = new AbortController();

  // LOOP-04: Beide Signale registrieren. Beide rufen shutdown.
  // Mehrfaches Senden (Doppel-Ctrl-C) wird durch sich selbst entfernende
  // Handler abgesichert: Nach erstem Signal kein zweites mehr verarbeiten,
  // sonst überlappen mehrere shutdown-Calls.
  let shuttingDown = false;
  const onSignal = (signal: NodeJS.Signals): void => {
    if (shuttingDown) {
      log.warn({ signal }, "[shutdown] already in progress, ignoring");
      return;
    }
    shuttingDown = true;
    void shutdown(signal, abort, sink);
  };
  process.on("SIGTERM", onSignal);
  process.on("SIGINT", onSignal);

  // LOOP-05: Last-Word-Log für unhandled Async-Errors. pino.final macht den
  // Write synchron — sonst gehen die letzten Log-Zeilen bei sofortigem exit
  // verloren (Pitfall 17). exit(1), weil das ein Bug ist; launchd entscheidet
  // in Phase 4 via KeepAlive-Strategy über Restart.
  process.on("unhandledRejection", (reason) => {
    finalLog.fatal({ reason }, "[fatal] unhandled rejection");
    process.exit(1);
  });
  process.on("uncaughtException", (err) => {
    finalLog.fatal({ err }, "[fatal] uncaught exception");
    process.exit(1);
  });

  log.info({ intervalMs: 10_000 }, "[daemon] starting polling loop");
  runLoop(sink, getNowPlaying, abort);
}

// Top-Level-await wäre möglich (ESM), aber main().catch ist robuster:
// jeder Throw beim Start landet im finalLog, exit(1) für launchd-KeepAlive-Strategy.
main().catch((err) => {
  finalLog.fatal({ err }, "[fatal] daemon failed to start");
  process.exit(1);
});
```
